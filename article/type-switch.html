type switch

Created At：2023-06-02 10:39 Updated At：2023-06-02 12:01

针对 interface{} 类型的变量，我们需要针对这个变量真实的类型进行转换、处理，则可使用 type switch 进行进行判断、处理，如下：

package main

import (
    "fmt"
)

func main() {
    var i int
    fmt.Println(checkType(i))
}

func checkType(i interface{}) string {
    // i 参数为 interface{}, 可以传递任意类型
    
    // 通过 switch type 的形式，来进行类型判断
    // 用于针对变量类型不确定情况下，按照不同类型进行对应不同行为处理的场景
    switch i.(type) {
    case string:
        return "string"	
    case int:
        return "int"
    case bool:
        return "bool"
    default:
        return "unknown"
    }
}

针对已知类型，可直接使用 i.(类型) 进行转换，例如：

package main

import (
    "fmt"
)

type I interface {
    SayName() string
}

type User struct {
    Name string
}

// User 实现了 I 接口
func (u User) SayName() string {
    return u.Name
}

func (u User) SayHello() {
    fmt.Println("hello")
}

func main() {
    // 此时 i 类型是 I
    var i I = User{
        Name: "user",
    }

    call(i) // 正常输入，因为 i 实现了 I 接口

    // i.SayHello() // 直接执行 i.SayHello() 报错，因为 I 接口无 SayHello() 方法

    // 使用 变量.(类型) 进行转换后，b 类型为 User{}, 可进行 SayHello()
    b := i.(User)

    b.SayHello()
    
    i = i.(User)
    i.SayHello() // 也会报错，因为 i 的类型最开始就已经限定了是 I
}

func call(i I) {
    fmt.Println(i.SayName())
}
